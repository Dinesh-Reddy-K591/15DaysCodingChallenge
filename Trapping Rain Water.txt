public class Solution {
    public boolean hasCycle(ListNode head) {
        if(head==null){
            return false;
        }
        ListNode r=head;
        ListNode t=head;
        while(r!=null && r.next!=null){
            r=r.next.next;
            t=t.next;
            if(r==t){
                return true;
            }
        }
        return false;
    }
}
