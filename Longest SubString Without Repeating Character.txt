class Solution {
    public int lengthOfLongestSubstring(String s) {
       HashSet<Character>map=new HashSet<>();
       int l=0,r=0,max=0;
       while(r<s.length()){
        if(!map.contains(s.charAt(r))){
            map.add(s.charAt(r));
            max=Math.max(max,r-l+1);
            r++;
        }
        else{
            map.remove(s.charAt(l));
            l++;
        }
       }
       return max;
    }
}
